\indent In this section, first, we review the basic theory of profinite Dehn multi-twists (cf. \cite[\S 4]{CbTpI}). We then introduce the notion of Dehn-twist pointed stable curves type outer representations as a certain generalization of IPSC-type [cf. \cite[Definition 2.4, (i)]{NodNon}], SVA-type [cf. \cite[Definition 2.4, (ii)]{NodNon}] and SNN-type [cf. \cite[Definition 2.4, (iii)]{NodNon}] outer representations, and investigate their basic properties. Finally, we investigate the natural outer representations of semi-graph of anabelioids associated to the strata defined in the previous section, and in particular verify that the outer representations of inertia groups are of Dehn-twist pointed stable curves type.

\begin{defn}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Then we shall write
  \[
    \Dehn{\Gcal} \coloneqq \{ \alpha \in \Aut^{|\text{grph}|}(\Gcal) \mid \alpha_{\Gcal|_v} = \mathrm{id}_{\Gcal|_v} \text{ for any } v \in v(\Gcal) \}
  \]
  --- where we refer to \cite[Definition 2.1, (iii)]{CbTpI}, \cite[Definition 2.14, (ii)]{CbTpI}, \cite[Remark 2.5.1]{CbTpI}. We shall refer to an element of $\Dehn{\Gcal}$ as a \emph{profinite Dehn multi-twist} of $\Gcal$.
\end{defn}

\begin{defn}\label{SignedNodalGroup}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Write $\univG \to \Gcal$ for the universal [pro-]covering of $\Gcal$ corresponding to $\PiG$. Let $\univ{e} \in \Node{\univG}$, $e \coloneqq \univ{e}(\Gcal) \in \Node{\Gcal}$. Write $\univ{v_1}, \univ{v_2} \in \Vcal(\univ{e})$ for the two vertices of $\univG$ adjacent to $\univ{e}$. Let $M_1, M_2 \coloneqq \Pi_{\univ{e}}$ (respectively, $N_1, N_2 \coloneqq \Pi_e$). Write $\Delta_M$ (respectively, $\Delta_N$) be the image of the ``diagonal'' embedding $\Pi_{\univ{e}} \to M_1 \times M_2$; $x \mapsto (x, x)$ (respectively, $\Pi_e \to N_1 \times N_2$; $x \mapsto (x, x)$). We shall refer to $M_1 \times M_2 / \Delta_M$ (respectively, $N_1 \times N_2 / \Delta_N$) as the \emph{signed nodal group} associated to $\univ{e}$ (respectively, $e$), and denote it by $\sgnnodal{\univ{e}}$ (respectively, $\sgnnodal{e}$). Let $x \in \Pi_{\univ{e}}$ (respectively, $x \in \Pi_e$). We shall denote the image of $(x, 1) \in M_1 \times M_2$ (respectively, $(x, 1) \in N_1 \times N_2$) in $\sgnnodal{\univ{e}}$ (respectively, $\sgnnodal{e}$) by $[x, \univ{v_1}]$ (respectively, $[x, v_1]$), and the image of $(1, x) \in M_1 \times M_2$ (respectively, $(1, x) \in N_1 \times N_2$) in $\sgnnodal{\univ{e}}$ (respectively, $\sgnnodal{e}$) by $[x, \univ{v_2}]$ (respectively, $[x, v_2]$).
\end{defn}

\begin{rem}
  In the notation of Definition \ref{SignedNodalGroup}, the signed nodal group $\sgnnodal{\univ{e}}$ associated to $\univ{e}$ can be naturally identified with the signed nodal group $\sgnnodal{e}$ associated to $e$.
\end{rem}

\begin{rem}\label{CyclotomesAndSignedNodalGroup}
  By \cite[Corollary 3.9, (vi)]{CbTpI}, there exists a natural isomorphism $\sgnnodal{e} \cong \Cyclo_\Gcal$ (synchronization of cyclotomes), where we write $\Cyclo_\Gcal$ for the cyclotome associated to $\Gcal$ (cf. \cite[Definition 3.8, (i)]{CbTpI}).
\end{rem}

\begin{defn}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type.
  \begin{enumerate}
    \item Let $\alpha \in \Dehn{\Gcal}$ be a profinite Dehn multi-twist of $\Gcal$. Let $\univ{e} \in \Node{\univG}$, $e \coloneqq \univ{e}(\Gcal) \in \Node{\Gcal}$. We define an element of $\sgnnodal{\univ{e}}$ as follows.
  \begin{itemize}
    \item Let $\univ{v} \in \Vcal(\univ{e})$. Then there exists a unique lifting $\alpha[\univ{v}] \in \Aut(\PiG)$ of $\alpha$ such that $\alpha$ preserves $\Pi_{\univ{v}}$ and induces the identity on $\Pi_{\univ{v}}$. Here, letting $\univ{w} \in \Vcal(\univ{e})$ be the other vertex of $\univG$ adjacent to $\univ{e}$, $\alpha$ induces an inner automorphism of $\Pi_{\univ{w}}$. Therefore, there exists a unique element $\delta_{\alpha, \univ{v}} \in \Pi_{\univ{e}}$ such that $\alpha[\univ{v}]|_{\Pi_{\univ{w}}}$ coincides with the inner automorphism of $\Pi_{\univ{w}}$ arising from the conjugation action of $\delta_{\alpha, \univ{v}}$. We shall denote the element $[\delta_{\alpha, \univ{v}}, \univ{v}]$ of $\sgnnodal{\univ{e}}$ by $\delta_{\alpha, \univ{e}}$.
  \end{itemize}
  We note, in light of \cite[Lemma 4.6]{CbTpI}, that this definition is well-defined. We denote the morphism $\Dehn{\Gcal} \to \sgnnodal{\univ{e}}$ given by assigning $\alpha \mapsto \delta_{\alpha, \univ{e}}$ by $\Dpar{\univ{e}}$. Moreover, the composite morphism $\Dehn{\Gcal} \xrightarrow{\Dpar{\univ{e}}} \sgnnodal{\univ{e}} \cong \sgnnodal{e}$ depends only on $e$ (cf. \cite[Lemma 4.6]{CbTpI}), and we shall denote it by $\Dpar{e}$.
  \item We shall write \[\Dparall{\Gcal} \coloneqq \bigoplus_{e \in \Node{\Gcal}} \Dpar{e} : \Dehn{\Gcal} \to \bigoplus_{e \in \Node{\Gcal}} \sgnnodal{e}.\]
  \end{enumerate}
\end{defn}

\begin{lem}[Description of profinite Dehn multi-twists]\label{DescriptionOfDehnMultiTwists}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Then the following hold.
  \begin{enumerate}
    \item The morphism $\Dparall{\Gcal} : \Dehn{\Gcal} \to \bigoplus_{e \in \Node{\Gcal}} \sgnnodal{e}$ is isomorphism.
    \item Let $\alpha \in \Dehn{\Gcal}$ be a profinite Dehn multi-twist of $\Gcal$. For $e \in \Node{\Gcal}$, write $\delta_{\alpha, e} \coloneqq \Dpar{e}(\alpha)$. Let $\B(\Gcal)$ be the Galois category associated to $\Gcal$, $\alpha^* \colon \B(\Gcal) \to \B(\Gcal)$ the autoequivalence of $\B(\Gcal)$ induced by $\alpha$. Then $\alpha^*$ is isomorphic to the autoequivalence $\Psi_{\{\delta_{\alpha, e}\}_{e \in \Node{\Gcal}}}$ of $\B(\Gcal)$ given as follows:
    \begin{itemize}
      \item Let $\{S_v, \phi_e\} \in \B(\Gcal)$, where $S_v \in \B(\Gcal|_v)$ for each vertex $v$; for each node $e$, with branches $b_1$, $b_2$ abutting to vertices $v_1$, $v_2$, respectively, $\phi_e \colon \{(b_1)_*\}^*S_{v_1} \xrightarrow{\sim} \{(b_2)_*\}^*S_{v_2}$ is an isomorphism in $\B(\Gcal|_e)$. We define $\Psi_{\{\delta_{\alpha, e}\}_{e \in \Node{\Gcal}}}(\{S_v, \phi_e\})$ as follows. For each node $e$, we shall write $\delta_{\alpha, v_1}$ for the [unique] element of $\Pi_{\univ{e}}$ such that $[\delta_{\alpha, v_1}, \univ{v_1}] = \delta_{\alpha, e}$. Then we define $\Psi_{\{\delta_{\alpha, e}\}_{e \in \Node{\Gcal}}}(\{S_v, \phi_e\})$ to be the object $\{S_v, \phi'_e\}$ of $\B(\Gcal)$, where for each node $e$, with branches $b_1$, $b_2$ abutting to vertices $v_1$, $v_2$, respectively, $\phi'_e$ is the isomorphism in $\B(\Gcal|_e)$ obtained as the composite of the following isomorphisms: we shall refer to the composite isomorphism $\phi_e \circ \mathsf{action}_{\delta_{\alpha, v_1}} \colon \{(b_1)_*\}^*S_{v_1} \xrightarrow{\sim} \{(b_2)_*\}^*S_{v_2}$, $\phi'_e$, where $\mathsf{action}_{\delta_{\alpha, v_1}}$ is the isomorphism of $\{(b_1)_*\}^*S_{v_1}$ arising from the action of $\delta_{\alpha, v_1}$.
    \end{itemize}
  \end{enumerate}
\end{lem}
\begin{proof}
  Assertion (i) follows immediately from \cite[Theorem 4.8, (iv)]{CbTpI} and Remark \ref{CyclotomesAndSignedNodalGroup}.

  Assertion (ii) follows immediately from the various definitions involved.
\end{proof}

\begin{defn}\label{DefnDTPSCtype}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Let $J$ be a profinite group which acts continuously on $\Gcal$ [i.e., we are given a continuous homomorphism $\phi \colon J \to \Aut(\Gcal)$].
  \begin{enumerate}
    \item We shall write $\PiG^J$ for the profinite group $\PiG^\Aut \times_{\Aut(\Gcal)} J$ (cf. \cite[Definition 2.3]{CmbGC}).
    \item We shall say that the representation $\phi \colon J \to \Aut(\Gcal)$, or the outer representation $J \to \Out(\PiG)$ induced by $\phi$, is of \emph{Dehn-twist pointed stable curves type} if the inequality $\phi(J) \subset \Dehn{\Gcal}$ holds. We abbreviate the term ``Dehn-twist pointed stable curves type'' to ``\emph{DTPSC-type}''.
    \item Let $r$ be an integer. We shall say that the representation $\phi \colon J \to \Aut(\Gcal)$, or the outer representation $J \to \Out(\PiG)$ induced by $\phi$, is of \emph{Dehn-twist pointed stable curves type of rank $r$} if the representation is of DTPSC-type, and $J$ is isomorphic to the free pro-$\Sigma$ abelian group of rank $r$. We abbreviate the term ``Dehn-twist pointed stable curves type of rank $r$'' to ``\emph{DTPSC-type of rank $r$}''.
    \item Let $J'$ be a subgroup of $J$. We shall refer to the representation $J' \to \Aut(\Gcal)$ obtained by restricting $\phi$ to $J'$ as the \emph{restriction} of $\phi$ to $J'$.
    \item Let $e \in \Node{\Gcal}$. Assume that $\phi$ is of DTPSC-type. Then we shall say that the representation $\phi$ is \emph{degenerate at $e$} if the image of the composite morphism $J \xrightarrow{\phi} \Dehn{\Gcal} \xrightarrow{\Dpar{e}} \sgnnodal{e}$ is not open. We shall say that $\phi$ is \emph{non-degenerate at $e$} if $\phi$ is not degenerate at $e$. We shall say that $\phi$ is \emph{non-degenerate} if $\phi$ is non-degenerate at $e$ for any $e \in \Node{\Gcal}$.
    \item Assume that $\phi$ is of DTPSC-type. Let $D$ be a subset of $\Node{\Gcal}$. Assume that $\phi$ is degenerate at $e$ for any $e \in D$. Write $\Gcal_{\leadsto D}$ for the generization of $\Gcal$ with respect to $D$ (cf. \cite[Definition 2.8]{CbTpI}). Then, by \cite[Theorem 4.8, (ii)]{CbTpI}, the representation $\phi$ induces a natural representation $J \to \Dehn{\Gcal_{\leadsto D}} \subset \Aut(\Gcal_{\leadsto D})$. We shall refer to the representation $J \to \Aut(\Gcal_{\leadsto D})$ as the \emph{generization} of $\phi$ with respect to $D$.
  \end{enumerate}
\end{defn}

\begin{rem}\label{RemarkUniqueGenerization}
  We note that every outer representation of DTPSC-type has an [unique] generization which is non-degenerate.
\end{rem}

\begin{rem}
  Assume $\# \Sigma = 1$. Then, it follows immediately from the various definitions involved that the image of the morphism $J \to \sgnnodal{e}$ is not open if and only if the morphism $J \to \sgnnodal{e}$ is trivial.
\end{rem}

\begin{rem}
  We note that the notion of DTPSC-type is a certain generalization of the notion of SVA-type, introduced in \cite[Definition 2.4, (ii)]{NodNon}, and the property of being non-degenerate is a certain generalization of the property of being of NN-type, introduced in \cite[Definition 2.4, (iii)]{NodNon}.
\end{rem}

\begin{defn}\label{DefnDecompositionGroupInertiaGroup}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Let $J$ be a profinite group which acts continuously on $\Gcal$. Assume that the representation $\phi \colon J \to \Aut(\Gcal)$ is of DTPSC-type. In light of Definition \ref{DefnDTPSCtype}, we have an exact sequence
  \[
    1 \to \PiG \to \PiG^J \to J \to 1.
  \]
  \begin{enumerate}
    \item Let $v \in \Vert{\Gcal}$ be a vertex and let $\Pi_v \subset \PiG$ be a verticial subgroup of $\PiG$ associated to $v$. Then we shall write
    \[
      D_v \coloneqq N_{\PiG^J}(\Pi_v) \subset \PiG^J \ \mathrm{(respectively, \ } I_v \coloneqq Z_{\PiG^J}(\Pi_v) \subset \PiG^J \mathrm{)}
    \]
    and refer to $D_v$ (respectively, $I_v$) as a \emph{decomposition subgroup} (respectively, \emph{inertia subgroup}) of $\PiG^J$ associated to $v$, or, alternatively, the decomposition subgroup (respectively, inertia subgroup) of $\PiG^J$ associated to the verticial subgroup $\Pi_v \subset \PiG$. If, moreover, the verticial subgroup $\Pi_v$ is the verticial subgroup associated to an element $\univ{v} \in \Vert{\Gcal}$, then we shall write $D_{\univ{v}} \coloneqq D_v$ (respectively, $I_{\univ{v}} \coloneqq I_v$) and refer to $D_{\univ{v}}$ (respectively, $I_{\univ{v}}$) as the decomposition subgroup (respectively, inertia subgroup) of $\PiG^J$ associated to $\univ{v}$.
    \item Let $c \in \Cusp{\Gcal}$ be a cusp and let $\Pi_c \subset \PiG$ be a cuspidal subgroup of $\PiG$ associated to $c$. Then we shall write
    \[
      D_c \coloneqq N_{\PiG^J}(\Pi_c) \subset \PiG^J \ \mathrm{(respectively, \ } I_c \coloneqq \Pi_c \subset \PiG^J \mathrm{)}
    \]
    and refer to $D_c$ (respectively, $I_c$) as a \emph{decomposition subgroup} (respectively, \emph{inertia subgroup}) of $\PiG^J$ associated to $c$, or, alternatively, the decomposition subgroup (respectively, inertia subgroup) of $\PiG^J$ associated to the cuspidal subgroup $\Pi_c \subset \PiG$. If, moreover, the cuspidal subgroup $\Pi_c$ is the cuspidal subgroup associated to an element $\univ{c} \in \Cusp{\Gcal}$, then we shall write $D_{\univ{c}} \coloneqq D_c$ (respectively, $I_{\univ{c}} \coloneqq I_c$) and refer to $D_{\univ{c}}$ (respectively, $I_{\univ{c}}$) as the decomposition subgroup (respectively, inertia subgroup) of $\PiG^J$ associated to $\univ{c}$.
    \item Let $e \in \Node{\Gcal}$ be an node and let $\Pi_e \subset \PiG$ be an nodal subgroup of $\PiG$ associated to $e$. Then we shall write
    \[
      D_e \coloneqq N_{\PiG^J}(\Pi_e) \subset \PiG^J \ \mathrm{(respectively, \ } I_e \coloneqq Z_{\PiG^J}(\Pi_e) \subset \PiG^J \mathrm{)}
    \]
    and refer to $D_e$ (respectively, $I_e$) as a \emph{decomposition subgroup} (respectively, \emph{inertia subgroup}) of $\PiG^J$ associated to $e$, or, alternatively, the decomposition subgroup (respectively, inertia subgroup) of $\PiG^J$ associated to the nodal subgroup $\Pi_e \subset \PiG$. If, moreover, the nodal subgroup $\Pi_e$ is the nodal subgroup associated to an element $\univ{e} \in \Node{\Gcal}$, then we shall write $D_{\univ{e}} \coloneqq D_e$ (respectively, $I_{\univ{e}} \coloneqq I_e$) and refer to $D_{\univ{e}}$ (respectively, $I_{\univ{e}}$) as the decomposition subgroup (respectively, inertia subgroup) of $\PiG^J$ associated to $\univ{e}$.
  \end{enumerate}
\end{defn}

\begin{rem}
  We note here that while the terminology in Definition \ref{DefnDecompositionGroupInertiaGroup} may lack a certain internal consistency, preference was given to maintaining compatibility with the terminology used in \cite[Definition 2.2]{NodNon}.
\end{rem}

\begin{lem}[Basic Properties of Decomposition and Inertia Subgroups]\label{BasicPropertiesOfDecompositionAndInertiaSubgroups}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Let $J$ be a profinite group which acts continuously on $\Gcal$. Assume that the representation $\phi \colon J \to \Aut(\Gcal)$ is of DTPSC-type. Then the following hold.
  \begin{enumerate}
    \item Let $\univ{v} \in \Vert{\univG}$. Then the following equalities hold.
    \begin{itemize}
      \item $\PiG \cap I_{\univ{v}} = 1$.
      \item $\PiG \cap D_{\univ{v}} = \Pi_{\univ{v}}$.
    \end{itemize}
    Moreover, we have an exact sequence
    \[
      1 \to \Pi_{\univ{v}} \to D_{\univ{v}} \to J \to 1,
    \]
    and the morphism $I_{\univ{v}} \to J$ induced by the inclusion $I_{\univ{v}} \subset D_{\univ{v}}$ is an isomorphism. Thus, $D_{\univ{v}}$ is naturally isomorphic to the product $\Pi_{\univ{v}} \times I_{\univ{v}}$, or $\Pi_{\univ{v}} \times J$.
    \item Let $\univ{e} \in \Node{\univG}$, $\univ{v} \in \Vcal(\univ{e})$. Then the followings hold.
    \begin{itemize}
      \item $I_{\univ{v}} \subset I_{\univ{e}}$.
      \item $\PiG \cap I_{\univ{e}} = \Pi_{\univ{e}}$.
      \item $D_{\univ{e}} = I_{\univ{e}}$.
    \end{itemize}
    Moreover, we have an exact sequence
    \[
      1 \to \Pi_{\univ{e}} \to I_{\univ{e}} \to J \to 1,
    \]  
    and the morphism $I_{\univ{v}} \times \Pi_{\univ{e}} \to I_{\univ{e}}$ induced by the inclusions $I_{\univ{v}} \subset I_{\univ{e}}$ is an isomorphism.
    \item Let $\univ{c} \in \Cusp{\univG}$, $\univ{v} \in \Vcal(\univ{c})$. Then the following equality holds.
    \begin{itemize}
      \item $I_{\univ{v}} \subset D_{\univ{c}}$.
      \item $\PiG \cap D_{\univ{c}} = \Pi_{\univ{c}}$.
    \end{itemize}
    Moreover, we have an exact sequence
    \[
      1 \to \Pi_{\univ{c}} \to D_{\univ{c}} \to J \to 1,
    \]
    and the morphism $I_{\univ{v}} \times \Pi_{\univ{c}} \to D_{\univ{c}}$ induced by the inclusions $I_{\univ{v}} \subset D_{\univ{c}}$ is an isomorphism.
    \item Let $\univ{e} \in \Node{\univG}$, $\univ{v} \in \Vcal(\univ{e})$. (respectively, $\univ{c} \in \Cusp{\univG}$, $\univ{v} \in \Vcal(\univ{c})$. Then the inequality $D_{\univ{e}} \subset D_{\univ{v}}$ (respectively, $D_{\univ{c}} \subset D_{\univ{v}}$) holds.
  \end{enumerate}
\end{lem}
\begin{proof}
  The former portion of the assertion (i) follows immediately from the commensurably terminality of $\Pi_{\univ{v}}$ in $\PiG$ and the various definitions involved. To verify the latter portion of the assertion (i), it suffices to show that the morphism $I_{\univ{v}} \to J$ induced by the inclusion $I_{\univ{v}} \subset D_{\univ{v}}$ is an isomorphism. We note, by the equation $\PiG \cap I_{\univ{v}} = 1$, that the morphism $I_{\univ{v}} \to J$ is injective. Now, we take a morphism $s \colon J \to \mathrm{Aut}(\PiG)$ in a following manner: for $j \in J$, we define $s(j)$ for the unique lifting of $\phi(j)$ to an element of $\Aut(\PiG)$ which preserves $\Pi_{\univ{v}}$ and induces the identity on $\Pi_{\univ{v}}$. Then, by the definitions involved, the image of $s$ is contained in $I_{\univ{v}}$. Moreover, the composite morphism $J \xrightarrow{s} I_{\univ{v}} \to J$ is obviously the identity morphism. Thus, the morphism $I_{\univ{v}} \to J$ is an isomorphism, and the assertion (i) follows.
  
  The first two formulas in the assertion (ii) follow immediately from the commensurably terminality of $\Pi_{\univ{e}}$ in $\PiG$ and the various definitions involved. The remaining portion of the assertion (ii) follows immediately from the former two formulas in the assertion (ii) and the various definitions involved.

  The former portion of the assertion (iii) follows immediately from the commensurably terminality of $\Pi_{\univ{c}}$ in $\PiG$ and the various definitions involved. The remaining portion of the assertion (iii) follows immediately from the former portion of the assertion (iii) and the various definitions involved. The assertion (iv) follows immediately from the assertion (ii) and the assertion (iii).
\end{proof}

\begin{lem}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Let $J$ be a profinite group which acts continuously on $\Gcal$. Assume that the representation $\phi \colon J \to \Aut(\Gcal)$ is of DTPSC-type. Let $\univ{e} \in \Node{\univG}$. Let $\univ{v_1}, \univ{v_2} \in \Vcal(\univ{e})$ be the two vertices of $\univG$ adjacent to $\univ{e}$. Then the following hold.
  \begin{enumerate}
    \item In light of Lemma \ref{BasicPropertiesOfDecompositionAndInertiaSubgroups}, we have an isomorphism $I_{\univ{v_1}} \times \Pi_{\univ{e}} \to I_{\univ{v_2}} \times \Pi_{\univ{e}}$ induced by the inclusions $I_{\univ{v_1}} \subset I_{\univ{e}}$, $I_{\univ{v_2}} \subset I_{\univ{e}}$ and $\Pi_{\univ{e}} \subset I_{\univ{e}}$. Then the followings hold.
    \begin{itemize}
      \item The morphism $I_{\univ{v_1}} \to I_{\univ{v_2}}$ induced by the isomorphism $I_{\univ{v_1}} \times \Pi_{\univ{e}} \to I_{\univ{v_2}} \times \Pi_{\univ{e}}$ is an isomorphism, which is induced by the isomorphisms $I_{\univ{v_1}} \to J$ and $I_{\univ{v_2}} \to J$ given in Lemma \ref{BasicPropertiesOfDecompositionAndInertiaSubgroups}.
      \item The morphism $I_{\univ{v_1}} \to \Pi_{\univ{e}}$ coincides with the composite morphism $I_{\univ{v_1}} \to J \xrightarrow{\phi} \Dehn{\Gcal} \xrightarrow{\Dpar{\univ{e}}} \sgnnodal{\univ{e}} \to \Pi_{\univ{e}}$, where the morphism $\sgnnodal{\univ{e}} \to \Pi_{\univ{e}}$ is the morphism given by $[x, \univ{v_1}] \mapsto x$ for $x \in \Pi_{\univ{e}}$.
      \item The morphism $\Pi_{\univ{e}} \to I_{\univ{v_2}}$ induced by the isomorphism $I_{\univ{v_1}} \times \Pi_{\univ{e}} \to I_{\univ{v_2}} \times \Pi_{\univ{e}}$ is the zero morphism.
      \item The morphism $\Pi_{\univ{e}} \to \Pi_{univ{e}}$ induced by the isomorphism $I_{\univ{v_1}} \times \Pi_{\univ{e}} \to I_{\univ{v_2}} \times \Pi_{\univ{e}}$ is the identity morphism.
    \end{itemize}
    \item Assume that the [outer] representation $\phi$ is non-degenerate at $\univ{e}$. Then the morphism $I_{\univ{v_1}} \times I_{\univ{v_2}} \to I_{\univ{e}}$ has open image.
  \end{enumerate}
\end{lem}
\begin{proof}
  The assertion (i) follows immediately from the proof of Lemma \ref{BasicPropertiesOfDecompositionAndInertiaSubgroups} and the various definitions involved. The assertion (ii) follows immediately from the assertion (i) and the various definitions involved.
\end{proof}

\begin{defn}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Let $J$ be a profinite group which acts continuously on $\Gcal$. Assume that the representation $\phi \colon J \to \Aut(\Gcal)$ is of DTPSC-type. Then we shall denote by \[\Gcalrep{J}\] the connected semi-graph of anabelioids defined as follows.
  \begin{itemize}
    \item The underlying semi-graph of $\Gcalrep{J}$ is the underlying semi-graph of $\Gcal$.
    \item The anabelioid corresponding to a vertex $v \in \Vert{\Gcal}$ (respectively, a cusp $c \in \Cusp{\Gcal}$; a node $e \in \Node{\Gcal}$) is the connected anabelioid determined by the decomposition subgroup, regarded up to inner automorphism, $D_v \subset \PiG^J$ (respectively, $D_c \subset \PiG^J$; $D_e \subset \PiG^J$).
    \item For $v \in \Vcal(e)$, the associated morphism of anabelioids is the morphism determined by the natural inclusion $D_e \subset D_v$.
  \end{itemize}
\end{defn}

\begin{lem}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Let $J$ be a profinite group which acts continuously on $\Gcal$. Assume that the representation $\phi \colon J \to \Aut(\Gcal)$ is of DTPSC-type. Then, the natural outer isomorphism $\Pi_{\Gcalrep{J} \to \PiG^J}$, induced by the various inclusions $D_v \subset \PiG^J$ (respectively, $D_c \subset \PiG^J$; $D_e \subset \PiG^J$) for $v \in \Vert{\Gcal}$ (respectively, $c \in \Cusp{\Gcal}$; $e \in \Node{\Gcal}$), is an isomorphism.
\end{lem}
\begin{proof}
  First, we observe that the decomposition subgroup $D_z$ --- where $z \in \VCN{\Gcal}$ --- is an extension of $J$ by $\Pi_z$. Now it is easily verified that the profinite Galois covering of $\Gcalrep{J}$ determined by the various quotients $D_z \to J$ is isomorphic to $\Gcal$ [as a pro-object of the category of semi-graphs of anabelioids]; thus, we obtain an exact sequence
  \[
    1 \to \PiG \to \Pi_{\Gcalrep{J}} \to J \to 1.
  \]
  On the other hand, it follows from the construction of this profinite covering $\Gcal \to \Gcalrep{J}$, together with the definition of the homomorphism $\Pi_{\Gcalrep{J}} \to \PiG^J$, that the composite $\PiG \to \Pi_{\Gcalrep{J}} \to \PiG^J$ coincides with the natural inclusion $\PiG \to \PiG^J$. Thus, the bijectivity of the homomorphism $\Pi_{\Gcalrep{J}} \to \PiG^J$ follows from the ``Five Lemma''. This completes the proof of the lemma.
\end{proof}

\begin{defn}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Let $J$ be a profinite group which acts continuously on $\Gcal$. Let $r$ be an integer. Assume that the representation $\phi \colon J \to \Aut(\Gcal)$ is of DTPSC-type of rank $r$. Then we shall say that the representation $\phi$ is \emph{of multiple inertia pointed stable curves type} if the following conditions hold:
  \begin{itemize}
    \item there is a set of subgroups $\{J_i\}_{i \in \Lambda}$ such that, for each $i \in \Lambda$, the restriction of $\phi$ to the subgroup $J_i$ has a generization which is of IPSC-type (cf. \cite[Definition 2.4, (i)]{NodNon}, Remark \ref{RemarkUniqueGenerization}), and $J$ is generated by the subgroups $J_i$.
  \end{itemize}
  We abbreviate the term ``multiple inertia pointed stable curves type'' to ``\emph{MIPSC-type}''.
\end{defn}

\begin{lem}
  Let $\Sigma$ be a nonempty set of prime numbers, and let $\Gcal$ be a semi-graph of anabelioids of pro-$\Sigma$ PSC-type. Let $J$ be a profinite group which acts continuously on $\Gcal$. Assume that the representation $\phi \colon J \to \Aut(\Gcal)$ is of MIPSC-type. Then the representation $\phi$ is of DTPSC-type.
\end{lem}
\begin{proof}
  The assertion follows immediately from \cite[Remark 2.4.2]{NodNon} and the various definitions involved.
\end{proof}
